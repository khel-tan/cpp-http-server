108 };
109 auto keys = std::views::keys(data);
110 auto values = std::views::values(data);
111 std::string parameters = std::accumulate(

113 parametersFold);
114 std::string arguments
115 = std::accumulate(values.begin(), values.end(),
116 std::string{}, argumentsFold);
